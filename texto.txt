import pyautogui
import keyboard
import time

print("Movimente o mouse...")
print("Pressione F8 para salvar posição")
print("Pressione ESC para sair\n")

posicoes = []

while True:
    x, y = pyautogui.position()
    print(f"\rPosição atual: X={x} Y={y}", end="")

    if keyboard.is_pressed("F8"):
        posicoes.append((x, y))
        print(f"\nSalvo: X={x} Y={y}")
        time.sleep(0.5)  # evita salvar várias vezes seguidas

    if keyboard.is_pressed("esc"):
        print("\n\nEncerrando...")
        break

    time.sleep(0.1)

print("\nPosições gravadas:")
for i, (x, y) in enumerate(posicoes):
    print(f"{i+1}: X={x}, Y={y}")
